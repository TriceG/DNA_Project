import random

class Nucleotide():
    def __init__(self, xPos, yPos, l, h, type, speed, colour):
        self.x = xPos
        self.y = yPos
        self.l = l
        self.h = h
        self.type = type
        self.speed = speed
        self.colour = colour
        
    def display(self):
        fill(self.colour, self.colour, self.colour)
        rect(self.x,self.y,self.l,self.h)
        
    def move(self):
        self.x += self.speed
